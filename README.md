```bash
export URI_DE_CE_REPO=git@github.com:Jean-Baptiste-Lasselle/camunda-survey.git
mkdir -p ./camunda-survey/ 
git clone $URI_DE_CE_REPO ./camunda-survey/
cd ./camunda-survey/
chmod +x ./operations.sh

```
